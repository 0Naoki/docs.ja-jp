## <a name="example"></a><span data-ttu-id="62abd-108">例</span><span class="sxs-lookup"><span data-stu-id="62abd-108">Example</span></span>

```csharp  
// cs0314.cs

public class ClassConstraint { }

{ }  
  
public class B<T> : A<T> //CS0314  
{ }  
  
// Try using this instead.

{ }  
```
