## <a name="example"></a><span data-ttu-id="b78e5-107">例</span><span class="sxs-lookup"><span data-stu-id="b78e5-107">Example</span></span>

public interface I  
{  
    [CLSCompliant(false)]  
    int M();   // CS3010

{  
    public int M()

    }

    }  
}
