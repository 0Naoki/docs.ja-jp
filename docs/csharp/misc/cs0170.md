 <span data-ttu-id="28f0c-107">次の例では CS0170 が生成されます。</span><span class="sxs-lookup"><span data-stu-id="28f0c-107">The following sample generates CS0170:</span></span>  
  
```csharp  
// CS0170.cs  
public struct error  
{  
   public int i;  
}  
  
public class MyClass  
{  
   public static void Main()  
   {  
      error e;  
      // uncomment the next line to resolve this error  
      // e.i = 0;  
      System.Console.WriteLine( e.i );   // CS0170 because   
                                         //e.i was never assigned  
   }  
}  
```
