## <a name="example"></a><span data-ttu-id="91e49-109">例</span><span class="sxs-lookup"><span data-stu-id="91e49-109">Example</span></span>

using System;  
public static class Extensions  
{  
    public static S ExtMethod(this S s)  
    {  
        return s;  
    }  
}

{  
}
