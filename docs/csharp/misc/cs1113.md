## <a name="example"></a><span data-ttu-id="0059a-109">例</span><span class="sxs-lookup"><span data-stu-id="0059a-109">Example</span></span>

```csharp  
using System;  
public static class Extensions  
{  
    public static S ExtMethod(this S s)  
    {  
        return s;  
    }  
}

{  
}

{  
    static int Main()  
    {

    }  
}  
```
