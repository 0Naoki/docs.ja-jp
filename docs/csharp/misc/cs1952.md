## <a name="example"></a><span data-ttu-id="e2792-107">例</span><span class="sxs-lookup"><span data-stu-id="e2792-107">Example</span></span>  
 <span data-ttu-id="e2792-108">次のコードでは CS1952 が生成されます。</span><span class="sxs-lookup"><span data-stu-id="e2792-108">The following code produces CS1952:</span></span>

using System;  
using System.Linq.Expressions;  
  
class Test  
{  
    public static int M(__arglist)  
    {  
        return 1;  
    }  
  
    static int Main()  
    {  
        Expression<Func<int, int>> f = x => Test.M(__arglist(x)); // CS1952

    }  
}
