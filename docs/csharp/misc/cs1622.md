 <span data-ttu-id="e6339-107">次の例では CS1622 が生成されます。</span><span class="sxs-lookup"><span data-stu-id="e6339-107">The following sample generates CS1622:</span></span>  
  
```  
// CS1622.cs  
// compile with: /target:library  
using System.Collections;  
  
class C : IEnumerable  
{  
   public IEnumerator GetEnumerator()  
   {  
      return (IEnumerator) this;  // CS1622  
      yield return this;   // OK  
   }  
}  
```
