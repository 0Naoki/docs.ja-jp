## <a name="example"></a><span data-ttu-id="d7018-109">例</span><span class="sxs-lookup"><span data-stu-id="d7018-109">Example</span></span>

{  
    IEnumerator IEnumerable.GetEnumerator()  
    {  
        yield return 0;
