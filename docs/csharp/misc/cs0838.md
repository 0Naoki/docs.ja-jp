## <a name="example"></a><span data-ttu-id="da7b5-107">例</span><span class="sxs-lookup"><span data-stu-id="da7b5-107">Example</span></span>  
 <span data-ttu-id="da7b5-108">次の例では CS0838 が生成されます。</span><span class="sxs-lookup"><span data-stu-id="da7b5-108">The following example generates CS0838:</span></span>

using System.Linq;  
using System.Linq.Expressions;  
  
namespace TestNamespace  
{  
    class Test  
    {  
        static int Main()  
        {

                () => new int[2, 2] { { 1, 2 }, { 3, 4 } }; // CS0838

            int[,] nums = new int[2, 2] { { 1, 2 }, { 3, 4 } };  
            Expression<Func<int[,]>> expr2 = () => nums;   
  
            return 1;  
        }  
    }  
}
