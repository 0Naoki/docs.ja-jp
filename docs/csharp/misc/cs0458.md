```csharp

using System;  
public  class Test   
{  
    public static void Main()  
    {  
        int a = 5;  
        int? b = a + null;    // CS0458  
        int? qa = 15;  
        b = qa + null;        // CS0458  
        b -= null;            // CS0458

        b = qa2 + null;       // CS0458  
        qa2 -= null;          // CS0458  
    }  
}
```
