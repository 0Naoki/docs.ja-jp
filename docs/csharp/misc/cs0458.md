## <a name="example"></a><span data-ttu-id="3d7b5-106">例</span><span class="sxs-lookup"><span data-stu-id="3d7b5-106">Example</span></span>

using System;  
public  class Test   
{  
    public static void Main()  
    {  
        int a = 5;  
        int? b = a + null;    // CS0458  
        int? qa = 15;  
        b = qa + null;        // CS0458  
        b -= null;            // CS0458  
        int? qa2 = null;  
        b = qa2 + null;       // CS0458  
        qa2 -= null;          // CS0458  
    }  
}
